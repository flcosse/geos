Carte interactive avec Dash et Plotly

  • Python
  • Dash
  • Plotly
  • Flask

Une série de graphiques et de cartes sont générée via Plotly. L'interactivité du site est assurée par Dash, l'application est hébergée sur Heroku.

@app.callback(
    Output(component_id="the_graph", component_property="figure"),
    [Input(component_id="my_checklist", component_property="value")],
)
def update_graph(options_chosen):
    dff = df.copy()
    dff = dff[dff["Origine"].isin(options_chosen)]
    dff["Génération estimée (gwh)"] = dff["Génération estimée (gwh)"].fillna(0)
    px.set_mapbox_access_token(open(".mapbox_token.txt").read())
    fig = px.scatter_mapbox(
        dff,
        lat=dff["latitude"],
        lon=dff["longitude"],
        color=dff["Origine"],
        size=dff["Génération estimée (gwh)"],
        size_max=30,
        zoom=3,
        color_discrete_map=c,
        custom_data=["Origine", "Génération estimée (gwh)", "name"],
        title="Sites de production d'électricité par type d'énergie",
        mapbox_style="light",
        center={"lat": 48.4889191993939, "lon": 16.793357414739376},
    )
    fig.update_traces(
        hovertemplate="
".join(
            [
                "Y : %{lat:.2f}",
                "X : %{lon:.2f}",
                "Génération : %{customdata[1]:.0f} GWh",
                "Nom : %{customdata[2]}",
            ]
        )
    )
    fig.update_layout(legend=dict(yanchor="top", y=1, xanchor="left", x=0, bgcolor='rgba(250,250,250,.8)'))
    fig.update_layout(margin={"r": 0, "l": 0, "b": 0, "t": 30})
    fig.update_layout(legend_itemdoubleclick=False)
    return fig


@app.callback(
    Output(component_id="graph4", component_property="figure"),
    [Input(component_id="slct_year", component_property="value")],
)
def update_graph(option_slctd):
    source = []
    dff = df.copy()
    dff = dff.dropna()
    dff = dff[dff["Country"] == option_slctd]
    dff = dff.groupby(["Origine"]).sum("capacity_mw").reset_index()
    perc = [(i / sum(dff["capacity_mw"]) * 100) for i in dff["capacity_mw"]]
    dff["Pourcentage"] = perc
    dff["Pourcentage"] = dff["Pourcentage"].astype(int)
    for i in dff["Origine"]:
        if i == "Nucléaire":
            source.append("Nucléaire")
        elif i in ("Gaz", "Pétrole", "Charbon"):
            source.append("Traditionnelles")
        else:
            source.append("Renouvelables")
    dff["Sources"] = source
    fig = px.histogram(
        dff,
        y="Pourcentage",
        x="Sources",
        color="Origine",
        color_discrete_map=c,
        title="Mix selon les capacités maximales de production",
        orientation="v",
    )
    fig.update_layout(
        xaxis_title="capacity_mw",
    )
    fig.update_layout(yaxis_title="", xaxis_title="Sources")
    fig.update_traces(hovertemplate="%{y:.0f}%")
    fig.update_layout(margin={"r": 0, "l": 0, "b": 0, "t": 45})
    fig.update_layout(
        {
            "plot_bgcolor": "rgb(248, 249, 250)",
            "paper_bgcolor": "rgb(248, 249, 250)",
        }
    )
    fig.layout.xaxis.fixedrange = True
    fig.layout.yaxis.fixedrange = True
    fig.update_yaxes(zeroline=True, zerolinewidth=1, zerolinecolor='lightgray')
    fig.update_yaxes(showgrid=True, gridwidth=0.2, gridcolor="lightgray")
    fig.update_layout(legend_itemdoubleclick=False)
    return fig

@app.callback(
    Output(component_id="graph2", component_property="figure"),
    [Input(component_id="slct_year", component_property="value")],
)
def plot(pays):
    dff = df.copy()
    dff = dff[dff["Country"] == pays]
    dff = dff.sort_values(by=["Génération estimée (gwh)"], ascending=False)
    dff = dff.dropna()
    dff["Rang"] = dff["Génération estimée (gwh)"].rank(ascending=False)
    fig = px.scatter(
        dff,
        x="Rang",
        y="Génération estimée (gwh)",
        color="Origine",
        color_discrete_map=c,
        title="Rang des unités de production",
        custom_data=["Origine", "Génération estimée (gwh)", "name"],
        log_y=True,
    )
    fig.update_layout(
        yaxis_title="Génération (GWh)",
    )
    fig.update_traces(
        hovertemplate="
".join(
            [
                "Rang : %{x}",
                "Génération : %{y:.2f} GWh",
                "Nom : %{customdata[2]}",
            ]
        )
    )
    fig.update_layout(margin={"r": 0, "l": 0, "b": 0, "t": 45})
    fig.update_layout(
        {
            "plot_bgcolor": "rgb(248, 249, 250)",
            "paper_bgcolor": "rgb(248, 249, 250)",
        }
    )
    fig.update_xaxes(showgrid=True, gridwidth=0.1, gridcolor="lightgray")
    fig.update_yaxes(showgrid=True, gridwidth=0.1, gridcolor="lightgray")
    fig.update_xaxes(zeroline=True, zerolinewidth=1, zerolinecolor='lightgray')
    fig.layout.xaxis.fixedrange = True
    fig.layout.yaxis.fixedrange = True
    fig.update_layout(legend_itemdoubleclick=False)
    return fig

@app.callback(
    Output(component_id="graph5", component_property="figure"),
    [Input(component_id="slct_year", component_property="value")],
)
def prop(pays):
    dff = df.copy()
    dff = dff[dff["Country"] == pays]
    dff = dff.groupby(["Origine"]).size().reset_index(name="Nombre")
    fig = px.pie(
        dff,
        values="Nombre",
        names="Origine",
        color="Origine",
        color_discrete_map=c,
        title="Nombre d'unités de production par type",
    )
    fig.update_layout(margin={"r": 0, "l": 0, "b": 0, "t": 45})
    fig.update_traces(hovertemplate="%{label} : %{value} (%{percent})")
    fig.update_traces(textinfo="value")
    fig.update_layout(
        {
            "plot_bgcolor": "rgb(248, 249, 250)",
            "paper_bgcolor": "rgb(248, 249, 250)",
        }
    )
    fig.update_layout(legend_itemdoubleclick=False)
    return fig
